284char *serialize_to_string(t_gen_list *list, t_element_to_string element_to_string)
285{
286 if (!list || !element_to_string)
287 return NULL;
288
289 // First, calculate total length needed
290 size_t total_len = 2; // '[' + ']'
291 t_node *current = list->head;
292 size_t count = 0;
293
294 while (current)
295 {
296 char *elem_str = element_to_string(current->value);
297 if (!elem_str)
298 return NULL;
299
300 total_len += str_len(elem_str);
301 if (current->next)
302 total_len += 2; // for ", "
303
304 free(elem_str);
305 current = current->next;
306 count++;
307 }
308
309 char *result = malloc(total_len + 1); // +1 for '\0'
310 if (!result)
311 return NULL;
312
313 size_t pos = 0;
314 result[pos++] = '[';
315 result[pos] = '\0';
316
317 current = list->head;
318 size_t i = 0;
319 while (current)
320 {
321 char *elem_str = element_to_string(current->value);
322 if (!elem_str)
323 {
324 free(result);
325 return NULL;
326 }
327
328 str_copy_at(result, elem_str, &pos);
329 free(elem_str);
330
331 if (current->next)
332 {
333 result[pos++] = ',';
334 result[pos++] = ' ';
335 result[pos] = '\0';
336 }
337
338 current = current->next;
339 i++;
340 }
341
342 result[pos++] = ']';
343 result[pos] = '\0';
344
345 return result;
346}

352char **serialize_to_string_array(t_gen_list *list, t_element_to_string element_to_string)
353{
354 if (!list || !element_to_string)
355 return NULL;
356
357 char **array = malloc((list->size + 1) * sizeof(char *));
358 if (!array)
359 return NULL;
360

362 size_t i = 0;
363
364 while (current)
365 {
366 array[i] = element_to_string(current->value);
367 if (!array[i])

370 free(array[j]);

375 current = current->next;

378 array[i] = NULL; // Null-terminate

389t_gen_list *deserialize_from_string_array(char **array, t_string_to_element string_to_element)
390{
391 if (!array || !string_to_element)
392 return NULL;
393
394 t_gen_list *list = (t_gen_list *)malloc(sizeof(t_gen_list));
395 if (!list) return NULL;
396
397 list->head = NULL;
398 list->size = 0;
399 t_node *last = NULL;
400
401 for (size_t i = 0; array[i] != NULL; i++) {
402 void *value = string_to_element(array[i]);
403 if (!value) goto error_cleanup;
404
405 t_node *node = (t_node *)malloc(sizeof(t_node));
406 if (!node) {
407 free(value);
408 goto error_cleanup;
409 }
410 node->value = value;
411 node->next = NULL;
412
413 if (!list->head)
414 list->head = node;
415 else
416 last->next = node;
417 last = node;
418 list->size++;
419 }
420
421 return list;
422
423error_cleanup:
424 t_node *tmp = list->head;
425 while (tmp) {
426 t_node *next = tmp->next;
427 free(tmp->value);
428 free(tmp);
429 tmp = next;
430 }
431 free(list);
432 return NULL;
