61void cd_execute(t_command *command, t_gen_list *environment)
62{
63 t_iter it;
64 char *old_directory;
65 char *target;
66 char *joined_path;
67
68 if (!command->args || command->args->size < 2)
69 {
70 fprintf(stderr, "cd: missing argument\n");
71 return;
72 }
73 old_directory = env_get(environment, "PWD");
74 it = iter_start(command->args);
75 iter_next(&it);
76 target = iter_next(&it);
77 if (access(target, F_OK) == 0)
78 {
79 if (chdir(target) == -1)
80 {
81 perror("cd");
82 free(old_directory);
83 return;
84 }
85 if (old_directory)
86 env_set(environment, old_directory, "OLDPWD");
87 env_set(environment, ft_strdup(target), "PWD");
88 free(old_directory);
89 return;
90 }
91 if (old_directory)
92 {
93 joined_path = ft_strjoin(old_directory, target);
94 if (access(joined_path, F_OK) == 0)
95 {
96 if (chdir(joined_path) == -1)
97 {
98 perror("cd");
99 free(joined_path);
100 free(old_directory);
101 return;
102 }
103 env_set(environment, old_directory, "OLDPWD");
104 env_set(environment, joined_path, "PWD");
105 free(joined_path);
106 free(old_directory);
107 return;
108 }
109 free(joined_path);
110 }
111 fprintf(stderr, "cd: no such file or directory: %s\n", target);
112 free(old_directory);
113}
