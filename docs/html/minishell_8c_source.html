29static char *history_checker(char *cmd)
30{
31 int i;
32
33 i = 0;
34 if (cmd[0] == '|' || cmd[ft_strlen(cmd)] == '|')
35 {
36 add_history(cmd);
37 return (NULL);
38 }
39 while (cmd[i])
40 {
41 if (ft_ispace(cmd[i]) == 0)
42 break ;
43 i++;
44 }
45 if (cmd[i] == '\0')
46 return (NULL);
47 return (cmd);
48}

50char *check_cmd(char *cmd)
51{
52 int i;
53 int b;
54
55 b = 0;
56 i = 0;
57 if (history_checker(cmd) == NULL)
58 return (NULL);
59 while (cmd[i])
60 {
61 if (cmd[i] == '"')
62 b = 1;
63 i++;
64 while (cmd[i] && b == 1)
65 {
66 if (cmd[i] == '"')
67 b = 0;
68 i++;
69 }
70 }
71 if (b != 0)
72 {
73 add_history(cmd);
74 return (NULL);
75 }
76 return (cmd);
77}
